Make syntax


# Compile
%.o : %.c
	cc -c $<

# Link
% : %.o
	cc -o $@ $<

Produce syntax


# Compile
[%{name}.o]
dep.c = %{name}.c
recipe = cc -c %{c}

# Link
[%{name}]
dep.o = %{name}.o
recipe = cc -o %{target} %{o}
A slight increase in verbosity, but an increase all the same. What does it buy us?
